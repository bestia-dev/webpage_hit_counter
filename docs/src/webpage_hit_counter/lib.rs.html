5// The `main.rs` has all the stdin and stdout.
6// The `lib.rs` must be in/out agnostic. That is the responsibility of the `main.rs`
7
8// The `lib.rs` does not have any real code. All the code is in modules in separate files.
9// The `lib.rs` has just the list of modules, it publishes module's functions or class for the caller
10// and it has some global stuff like the Error enum.
11
12// access to modules
13mod actix_mod;
14mod deadpool_mod;
15mod postgres_mod;
16
17// `pub use` allows the caller of the lib to access modules functions, structs or all(*)
18pub use actix_mod::*;
19pub use deadpool_mod::*;
20pub use postgres_mod::*;
21
22// The `main.rs` uses the `anyhow` error library.
23// The `lib.rs` uses the `thiserror` library.
24use thiserror::Error;
25
26/// all possible library errors for `thiserror`
27#[derive(Error, Debug)]
28pub enum LibraryError {
29    #[error("Name `{0}` is already uppercase.")]
30    Uppercase(String),
31    #[error("Unknown error.")]
32    Unknown,
33}
