7/// extract path info from "/webpage_hit_counter/get_svg_image/{webpage_id}.svg" url
8/// {webpage_id} - deserializes to a i32
9pub async fn get_svg_image(
10    db_pool: actix_web::web::Data<deadpool_postgres::Pool>,
11    path: actix_web::web::Path<i32>,
12    req: actix_web::HttpRequest,
13) -> impl actix_web::Responder {
14    let webpage_id = path.into_inner();
15
16    let now: chrono::DateTime<chrono::Utc> = chrono::Utc::now();
17    println!("{now} webpage_hit_counter/get_svg_image/{webpage_id}.svg");
18
19    // When I use preview in VSCode it makes hundreds of request to the server and
20    // this increments the counter enormously. I don't want that. I will check for Electron and avoid the increment.
21    let mut do_increment = true;
22    let user_agent = req.headers().get(actix_web::http::header::USER_AGENT);
23    if let Some(user_agent) = user_agent {
24        if let Ok(user_agent) = user_agent.to_str() {
25            println!("user_agent: {}", user_agent);
26            if user_agent.contains("Electron") {
27                // this is a VSCode preview and does not increment the hit counter
28                do_increment = false;
29            }
30        }
31    }
32
33    let hit_count = crate::postgres_mod::select_count(db_pool, webpage_id, do_increment).await.unwrap();
34
35    actix_web::HttpResponse::Ok()
36        .append_header(actix_web::http::header::ContentType(mime::IMAGE_SVG))
37        .append_header(actix_web::http::header::CacheControl(vec![
38            actix_web::http::header::CacheDirective::NoCache,
39            actix_web::http::header::CacheDirective::MaxAge(0u32),
40            actix_web::http::header::CacheDirective::NoStore,
41            actix_web::http::header::CacheDirective::SMaxAge(0u32),
42            actix_web::http::header::CacheDirective::ProxyRevalidate,
43        ]))
44        .append_header(("Pragma", "no-cache"))
45        .append_header(("Expires", "-1"))
46        .body(format!(
47            r##"
48<svg xmlns="http://www.w3.org/2000/svg" xmlns:xlink="http://www.w3.org/1999/xlink" 
49width="82" height="20">
50 <linearGradient id="smooth" x2="0" y2="100%">
51   <stop offset="0" stop-color="#bbb" stop-opacity=".1"/>
52   <stop offset="1" stop-opacity=".1"/>
53 </linearGradient>
54
55 <mask id="round">
56   <rect width="82" height="20" rx="3" ry="3" fill="#fff"/>
57 </mask>
58
59 <g mask="url(#round)">
60   <rect width="42" height="20" fill="#555555"/>
61   <rect x="42" width="52" height="20" fill="#79C83D"/>
62   <rect width="82" height="20" fill="url(#smooth)"/>
63 </g>
64
65 <g fill="#fff" text-anchor="middle" font-family="Verdana,DejaVu Sans,Geneva,sans-serif" font-size="11"> 
66   <text x="18" y="15" fill="#010101" fill-opacity=".3">Hits</text>
67   <text x="18" y="14" fill="#fff">Hits</text>
68   <text x="60" y="15" fill="#010101" fill-opacity=".3">{hit_count}</text>
69   <text x="60" y="14" fill="#fff">{hit_count}</text>
70 </g>
71</svg>    
72    "##
73        ))
74}
